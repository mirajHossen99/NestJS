------- install steps with Packages ------
    npm i -g @nestjs/cli
    nest new nestjs-lms
    npm install bcrypt
    npm add -D @types/bcrypt
    npm i @nestjs/mongoose mongoose
    npm i --save @nestjs/config
    npm install --save @nestjs/jwt
    npm i --save class-validator class-transformer
    

------ Create module -----
    nest g module auth
    nest g module user

------ Create controller -----
    nest g controller auth

------ Create service -----
    nest g service auth
    nest g service user

------ Create resource -----
    nest g resource course

------ project run ------
    npm run start



check version--
    nest -v
    nest --help
